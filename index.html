function generarDefinicionInteligente(query, item) { const grupo = item.grupo; const categoria = item.categoria; const denom = item.denominacion.toLowerCase(); let tipoGasto = ''; let uso = ''; // Clasificación por grupo (esto es CLAVE) switch (grupo) { case '0': tipoGasto = 'remuneración de personal'; uso = 'pago de sueldos, compensaciones o beneficios'; break; case '1': tipoGasto = 'bien de consumo'; uso = 'insumos que se consumen en la operativa diaria'; break; case '2': tipoGasto = 'servicio'; uso = 'prestaciones necesarias para el funcionamiento'; break; case '3': tipoGasto = 'bien de uso'; uso = 'equipamiento o activos duraderos'; break; case '5': tipoGasto = 'transferencia'; uso = 'pagos sin contraprestación directa'; break; default: tipoGasto = 'gasto público'; uso = 'necesidades operativas del organismo'; } // Afinar por contenido (esto simula "inteligencia") if (denom.includes('medic')) { uso = 'adquisición de medicamentos para atención de pacientes'; } if (denom.includes('limpieza')) { uso = 'mantenimiento higiénico del hospital'; } if (denom.includes('combustible')) { uso = 'funcionamiento de vehículos y traslados'; } if (denom.includes('equipo') || denom.includes('computadora')) { uso = 'infraestructura tecnológica del hospital'; } return `Corresponde a ${tipoGasto} dentro de ${categoria}. Se utiliza para ${uso}, asociado al objeto ${item.objeto}.`; }
